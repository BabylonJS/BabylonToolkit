# DemoBobber Script Component

This script component bobs a transform up and down. It demonstrates initializing the playground.babylon.js editor and attaching a script component to the sphere.

class Playground {
    public static CreateScene(engine: BABYLON.Engine, canvas: HTMLCanvasElement): BABYLON.Scene {
        // This creates a basic Babylon Scene object (non-mesh)
        var scene = new BABYLON.Scene(engine);

        // Initialize the babylon toolkit runtime and user input
        TOOLKIT.SceneManager.InitializePlayground(engine);

        // This creates and positions a free camera (non-mesh)
        var camera = new BABYLON.FreeCamera("camera1", new BABYLON.Vector3(0, 5, -10), scene);

        // This targets the camera to scene origin
        camera.setTarget(BABYLON.Vector3.Zero());

        // This attaches the camera to the canvas
        camera.attachControl(canvas, true);

        // This creates a light, aiming 0,1,0 - to the sky (non-mesh)
        var light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene);
        // Default intensity is 1. Let's dim the light a small amount
        light.intensity = 0.7;

        // Our built-in 'sphere' shape.
        var sphere = BABYLON.MeshBuilder.CreateSphere("sphere", {diameter: 2, segments: 32}, scene);

        // Move the sphere upward 2 units
        sphere.position.y = 2;

        // Attach demo script component to sphere
        var bobber = new PROJECT.DemoBobber(sphere, scene);
        bobber.amplitude = 0.5;
        bobber.frequency = 1.5;

        // Our built-in 'ground' shape.
        var ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 6, height: 6}, scene);

        return scene;
    }
}

// Demo bob the transform script component
namespace PROJECT {
    export class DemoBobber extends TOOLKIT.ScriptComponent {
        public amplitude: number = 0.25;
        public frequency: number = 1.0;
        public phaseOffset: number = 0.0;

        private startPosition: BABYLON.Vector3 = BABYLON.Vector3.Zero();
        private elapsedTime: number = 0.0;

        constructor(transform: BABYLON.TransformNode, scene: BABYLON.Scene, properties: any = {}, alias: string = "PROJECT.DemoBobber") {
            super(transform, scene, properties, alias);
        }

        protected start(): void {
            this.startPosition.copyFrom(this.transform.position);
        }

        protected update(): void {
            this.elapsedTime += this.getDeltaTime();

            const radians: number = (this.elapsedTime * this.frequency * Math.PI * 2.0) + this.phaseOffset;
            const offsetY: number = Math.sin(radians) * this.amplitude;

            this.transform.position.y = this.startPosition.y + offsetY;
        }
    }
    
    TOOLKIT.SceneManager.RegisterClass("PROJECT.DemoBobber", DemoBobber);  
}
