# DemoUserInput Script Component

This script component bobs a transform up and down. It demonstrates initializing the playground.babylon.js editor, enabling user input and attaching a script component to the sphere.

```
class Playground {
    public static CreateScene(engine: BABYLON.Engine, canvas: HTMLCanvasElement): BABYLON.Scene {
        // This creates a basic Babylon Scene object (non-mesh)
        var scene = new BABYLON.Scene(engine);

        // Initialize the babylon toolkit runtime and user input
        TOOLKIT.SceneManager.InitializePlayground(engine);
        TOOLKIT.InputController.EnableUserInput(engine, scene);

        // This creates and positions a free camera (non-mesh)
        var camera = new BABYLON.FreeCamera("camera1", new BABYLON.Vector3(0, 5, -10), scene);

        // This targets the camera to scene origin
        camera.setTarget(BABYLON.Vector3.Zero());

        // This creates a light, aiming 0,1,0 - to the sky (non-mesh)
        var light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene);
        // Default intensity is 1. Let's dim the light a small amount
        light.intensity = 0.7;

        // Our built-in 'box' shape.
        var box = BABYLON.MeshBuilder.CreateBox("box", {size: 1}, scene);

        // Move the box upward 2 units
        box.position.y = 2;

        // Attach demo script component to sphere
        var mover = new PROJECT.DemoUserInput(box, scene);
        mover.moveSpeed = 5.0;
        mover.rotateSpeed = 120;
        mover.useLocalSpace = true;
        mover.enableMouseLook = true;

        // Our built-in 'ground' shape.
        var ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 6, height: 6}, scene);

        return scene;
    }
}

// Demo move the transform script component
namespace PROJECT {
    export class DemoUserInput extends TOOLKIT.ScriptComponent {
        public moveSpeed: number = 5.0;
        public rotateSpeed: number = 120.0;
        public useLocalSpace: boolean = true;
        public enableMouseLook: boolean = false;

        private movement: BABYLON.Vector3 = BABYLON.Vector3.Zero();

        constructor(transform: BABYLON.TransformNode, scene: BABYLON.Scene, properties: any = {}, alias: string = "PROJECT.DemoUserInput") {
            super(transform, scene, properties, alias);
        }

        protected update(): void {
            const deltaTime: number = this.getDeltaTime();

            const inputX: number = TOOLKIT.InputController.GetUserInput(TOOLKIT.UserInputAxis.Horizontal);
            const inputZ: number = TOOLKIT.InputController.GetUserInput(TOOLKIT.UserInputAxis.Vertical);

            this.movement.set(inputX, 0.0, inputZ);

            if (this.movement.lengthSquared() > 1.0) {
                this.movement.normalize();
            }

            this.movement.scaleInPlace(this.moveSpeed * deltaTime);

            if (this.useLocalSpace === true) {
                this.transform.locallyTranslate(this.movement);
            } else {
                this.transform.position.addInPlace(this.movement);
            }

            if (this.enableMouseLook === true) {
                const mouseX: number = TOOLKIT.InputController.GetUserInput(TOOLKIT.UserInputAxis.MouseX);
                const yaw: number = mouseX * this.rotateSpeed * deltaTime;

                this.transform.rotate(BABYLON.Axis.Y, BABYLON.Tools.ToRadians(yaw), BABYLON.Space.LOCAL);
            }
        }
    }
    
    TOOLKIT.SceneManager.RegisterClass("PROJECT.DemoUserInput", DemoUserInput);  
}
```
