# DemoRotator Script Component

This script component rotates a transform around the up axis. It demonstrates initializing the playground.babylon.js editor and attaching a script component to the box.

        // This creates a basic Babylon Scene object (non-mesh)
        var scene = new BABYLON.Scene(engine);

        // Initialize the babylon toolkit runtime and user input

        var camera = new BABYLON.FreeCamera("camera1", new BABYLON.Vector3(0, 5, -10), scene);

        // This attaches the camera to the canvas

// Demo rotate the transform script component
namespace PROJECT {
  export class DemoRotator extends TOOLKIT.ScriptComponent {
    public message: string = "Hello Babylon Toolkit";

    constructor(transform: BABYLON.TransformNode, scene: BABYLON.Scene, properties: any = {}, alias: string = "PROJECT.DemoRotator") {
        super(transform, scene, properties, alias);
    }

    protected start(): void {
      console.log(this.message);
    }

    protected update(): void {
      let deltaTime: number = this.getDeltaTime();
      this.transform.rotate(BABYLON.Axis.Y, deltaTime);
    }
  }
  
  TOOLKIT.SceneManager.RegisterClass("PROJECT.DemoRotator", DemoRotator);  
}
